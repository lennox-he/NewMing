REPLACE_OR_CREATE:communist_china = {
	color = "communist_china_red"
	possible = {
		OR = {
			AND = {
				exists = c:CHI
				THIS = c:CHI
			}
			AND = {
				exists = c:BEI
				THIS = c:BEI
			}
		}
		has_law = law_type:law_council_republic
	}
}
