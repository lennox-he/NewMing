chinese_han_imperial_clothes_trigger = { #明皇帝
	exists = scope:character
	scope:character = {
		is_ruler = yes
		OR = {
			AND = {
				owner = {
					OR = {
						has_law = law_type:law_monarchy
						has_law = law_type:law_theocracy
					}
				}
				OR = {
					culture = {
						has_discrimination_trait = tradition_sinosphere
						has_discrimination_trait = language_mandarin
					}
					culture = { has_discrimination_trait = heritage_han }
				}
			}
			AND = {
				exists = c:CHI
				owner = c:CHI
				owner = {
					OR = {
						has_variable = jianbian
						NOT = { has_variable = is_qng_state }
					}
					has_law = law_type:law_monarchy
				}
			}
		}
	}
}
