	exists = scope:character
	scope:character = {
		is_ruler = yes

			has_variable = is_qng_state
			has_law = law_type:law_monarchy

			NOT = { has_variable = jianbian }
		}
		exists = c:CHI
		owner = c:CHI
	}
}

junfa_officers_clothes_pop_trigger = { #军阀
exists = scope:character
scope:character = {
	OR = {
		AND = {
			owner = {
				NOT = { has_law = law_type:law_monarchy }
				country_has_primary_culture = cu:han
				NOT = { has_variable = is_qng_state }
			}
			OR = { has_role = admiral has_role = general interest_group = { is_interest_group_type = ig_armed_forces } }
			commander_rank >= 0
		}
		AND = {
			owner = { has_government_type = gov_warlord_state }
			is_ruler = yes
		}
	}
}
}
